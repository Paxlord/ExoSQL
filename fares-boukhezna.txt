#19 SELECT UPPER(distrib.nom) as 'nom_distrib', count(film.id) as 'nb_films', sum(film.duree_min) as 'duree_totale' FROM film LEFT JOIN distrib ON distrib.id=film.distrib_id WHERE distrib_id IS NOT NULL GROUP BY distrib_id ORDER BY count(film.id) DESC, sum(film.duree_min) DESC;
